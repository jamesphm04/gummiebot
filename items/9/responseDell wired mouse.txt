         * @name - showAd
         * @desc - method to show the ad whenever the placeholder is almost viewable to the user
         * @param - event
         */

        function showAdHandler () {
            if (!document.hidden && isAdAlmostOrInViewport() && !refreshed) {

                // variable to avoid multiple ad refreshes
                refreshed = true;

                if(window.adUnits && window.adUnits.length > 0) {
                    preBidRequestBid();
                } else {
                    // we still load it via dfp if prebid ad units if not available
                    if (adSlot) {
                        googletag.pubads().refresh(adSlot, {changeCorrelator: false});
                    }
                }

                // when ad has been displayed, clear all the event listener associated to that ad.
                window.removeEventListener('scroll', scrollHandler);
                document.removeEventListener('visibilitychange', showAdHandler);
            }
        }

        /**
         * @name - runLazyLoad
         * @desc - method to initialize event listeners for the lazy loading of the ad
         */
        function runLazyLoad() {

            if (isAdAlmostOrInViewport() && !refreshed) {
                showAdHandler();
            } else {

                // event listeners for ad refresh when it's in viewport.
                // scroll - actively check the ad whether it is already almost viewable to the user while scolling
                window.addEventListener('scroll', scrollHandler);

                // visibilitychange - so that if the user is coming from a different tab it will execute "showAd" method.
                document.addEventListener('visibilitychange', showAdHandler);
            }
        }

        // create the element placeholder for the ad
        // this should be always called before refresh
        googletag.display(slotElementId);

        if (options.isLazyLoadEnabled) {

            runLazyLoad();

        } else {

            if (options.isPrebidEnabled) {

                if(window.adUnits && window.adUnits.length > 0) {
                    preBidRequestBid();
                }
            } else {

                if (adSlot) {
                    googletag.pubads().refresh(adSlot, {changeCorrelator: false});
                }
            }
        }

        // reference: https://developers.google.com/doubleclick-gpt/reference?visit_id=0-636507205932962590-3426384474&hl=en&rd=1#googletageventsslotonloadevent
        googletag.pubads().addEventListener('slotOnload', function(event) {
            if (event.slot && event.slot.getSlotElementId() !== slotElementId) return; // the event has fired, but for a different slot

            // only do it on a particular adslot, which is the billboard
            if (options.adSlotId === 'billboard') {
                jqCheck(function(){
                    reloadBxSlider($('[data-bxslider]'));
                });
            }

            // check if the jQuery is already loaded
            function jqCheck(fn) {
                var jqCheckCount = 0;
                var jqCheckInterval = setInterval(function(){
                    jqCheckCount += 1;
                    if (window.jQuery) {
                        clearInterval(jqCheckInterval);
                        if (typeof fn === 'function') {
                            fn();
                        }

                    }
                    // give up when reach maxium try - 3s
                    if (jqCheckCount === 6) {
                        clearInterval(jqCheckInterval);
                    }
                }, 500);
            }

            // check if the bxSlider is already loaded
            function reloadBxSlider($el) {
                if (jQuery.fn.bxSlider && $el.length && $el.data('gtauBxSlider')) {
                    $el.data('gtauBxSlider').reloadSlider();
                }
            }
        });

        googletag.pubads().addEventListener('slotRenderEnded', function(event) {

            if (event.slot && event.slot.getSlotElementId() !== slotElementId) return; // the event has fired, but for a different slot

            if (event.size === null) {
                if (document.querySelectorAll('#' + slotElementId + ' iframe').length > 0) {
                    document.querySelectorAll('#' + slotElementId + ' iframe')[0].style.visibility = "hidden";
                }
            }

            // if the element was hidden, and an ad has been returned
            // although apparently the google lib will always show this div when the ad has loaded anyway
            if (options.hideUntilLoaded && !event.isEmpty) {
                var adEl = document.getElementById(slotElementId);
                if (adEl && adEl.classList) adEl.classList.remove('c-hide');
            }

        });

    });

}
